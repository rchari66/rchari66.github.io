  • Type Assertion

    Type assertion takes a value and tries to create another version in the specified explicit type.

    v, ok := variable.(<type>)
    
-- 
type Counter struct {
    count int
}
    
func printString(value interface{}) {
    switch str := value.(type) {
    case string:
        fmt.Println(str)
    case Counter:
        fmt.Println(str.count)
    }
}
    
func main() {
    c := Counter{count: 10}
    var str string = "Hello world"
    
    printString(c)
    printString(str)
}
